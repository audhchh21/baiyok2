1. ``` git clone https://github.com/audhchh21/baiyok2.git ```

    APP_ENV=local
    APP_KEY=base64:MAIllVr+zMAKXdFhu4NmChAmFA/oZZfLT54q3Ga4EWk=
    APP_DEBUG=true
    APP_URL=http://localhost

    LOG_CHANNEL=stack

    DB_CONNECTION=mysql
    DB_HOST=127.0.0.1
    DB_PORT=3306
    DB_DATABASE=baiyok
    DB_USERNAME=root
    DB_PASSWORD=
